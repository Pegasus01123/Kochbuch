

\textbf {Zutaten:} \\
\begin{itemize}
	\item Suppengemüse
	\item 1 Tl Salz
	\item 2 l Wasser
\end{itemize}

\vspace* {2cm}

\textbf {Zubereitung:} \\
\begin{compactenum}
	\item Suppengemüse putzen und kleinschneiden
	\item Wasser aufgießen und salzen
	\item 30 Minuten kochen lassen und nochmal abschmecken
\end{compactenum}
